  function checkHangulNum(str){
    for(var j = 0; j < str.length; j++){
      var chr = str.substr(j,1);
      chr = escape(chr);
      if(j==0 && !isNaN(chr)){
        return false;
      } else {
        if(chr.charAt(1) == "u"){
          chr = chr.substr(2, (chr.length - 1));
          if((chr < "AC00") || (chr > "D7A3"))
            return false;
        } else if(isNaN(chr)) {
          return false;
        }
      }
    }
    return true;
  }
